267void calculointervalos (uint32_t *pun_ir_buffer, int32_t n_ir_buffer_length, int32_t *pn_heart_rate, int8_t *pch_hr_valid){
268
269 uint32_t un_ir_mean;
270 int32_t k;
271 int32_t n_th1, n_npks;
272 int32_t an_ir_valley_locs[40] ;
273 int32_t n_peak_interval_sum;
274
275 // calculates DC mean and subtract DC from ir
276 un_ir_mean = 0;
277 for (k = 0; k < n_ir_buffer_length; k++) {
278 un_ir_mean += pun_ir_buffer[k];
279 }
280 un_ir_mean = un_ir_mean / n_ir_buffer_length;
281
282 // remove DC and invert signal so that we can use peak detector as valley detector
283 for (k = 0; k < n_ir_buffer_length; k++){
284 an_x[k] = -1 * (pun_ir_buffer[k] - un_ir_mean);
285 }
286
287 // 4 pt Moving Average
288 for (k = 0; k < BUFFER_SIZE - MA4_SIZE; k++) {
289 an_x[k] = (an_x[k] + an_x[k + 1] + an_x[k + 2] + an_x[k + 3]) / 4;
290 }
291
292 // calculate threshold
293 n_th1 = 0;
294 for (k = 0; k < BUFFER_SIZE; k++) {
295 n_th1 += an_x[k];
296 }
297 n_th1 = n_th1 / BUFFER_SIZE;
298 if (n_th1 < 30) n_th1 = 30; // min allowed
299 if (n_th1 > 60) n_th1 = 60; // max allowed
300
301 for (k = 0; k < MAX_PEAKS; k++){
302 an_ir_valley_locs[k] = 0;
303 }
304
305 // since we flipped signal, we use peak detector as valley detector
306 maxim_find_peaks(an_ir_valley_locs, &n_npks, an_x, BUFFER_SIZE, n_th1, 4, MAX_PEAKS); // peak_height, peak_distance, max_num_peaks
307
308 n_peak_interval_sum = 0;
309 peak_count = 0;
310 if (n_npks >= 2) {
311 for (k = 1; k < n_npks; k++) {
312 int32_t interval = abs(an_ir_valley_locs[k] - an_ir_valley_locs[k - 1]);
313 peak_intervals[peak_count++] = interval;
314 n_peak_interval_sum += interval;
315 }
316 n_peak_interval_sum = n_peak_interval_sum / (n_npks - 1);
317 *pn_heart_rate = (int32_t)((FreqS * 60) / n_peak_interval_sum);
318 *pch_hr_valid = 1;
319 } else {
320 *pn_heart_rate = -999; // unable to calculate because # of peaks are too small
321 *pch_hr_valid = 0;
322 }

328void calculoVariabilidad (void){
329 // se usa la desviacion estandar de los intervalos RR (SDNN)
330
331 float mean = 0.0;
332 for (int i = 0; i < peak_count; i++) {
333 mean += peak_intervals[i];
334 }
335 mean /= peak_count;
336 for (int i = 0; i < peak_count; i++) {
337 HRV += pow(peak_intervals[i] - mean, 2);
338 }
339
340 HRV /= (peak_count - 1);
341 HRV = sqrt(HRV);
342 printf ("HRV= %f \n", HRV);
343}
